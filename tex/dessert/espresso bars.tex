\begin{recipe}{Espresso Chocolate Cookie Bars}{9\inch\X13\inch baking dish \hfill 350\0F \hfill 40 min.}{\textbf{Source:} Dustin, Brooklyn Community Chorus \hfill serves 24}
 \ing[3\fr12]{cups}{all-purpose flour}
 \ing[1]{tsp.}{baking soda}
 \ing[1]{tsp.}{baking powder}
 \ing[1]{tsp.}{salt}
 \ing[4]{tsp.}{instant coffee}
 Combine dry ingredients in a large mixing bowl.

 \ing[1]{cup}{butter}
 \ing[2]{cups}{dark brown sugar}
 \ing[\fr12]{cup}{white sugar}
 Melt the butter in a large mixing bowl. Add the brown and white sugar and mix until smooth.

 \ing[2]{}{eggs}
 \ing[4]{tsp.}{vanilla extract}
 Mix the eggs and vanilla into the butter and sugar.

 \ing[9]{oz.}{Nestle Espresso Chips (1 bag)}
 Add the dry ingredients gradually, stirring to combine. Mix in the espresso chips.

 \newstep Pour espresso bars into a greased baking dish. Bake for 40 min.\ or until golden. Cool 45--60 min., slice, and serve.
\end{recipe}
