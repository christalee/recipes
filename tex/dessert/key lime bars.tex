\begin{recipe}{Key Lime Cookie Bars}{9\inch\X13\inch baking dish \hfill 350\0F \hfill 40 min.}{\textbf{Source:} Dustin, Brooklyn Community Chorus \hfill serves 24}
 \ing[2\fr14]{cups}{all-purpose flour}
 \ing[\fr12]{tsp.}{baking soda}
 \ing[\fr12]{tsp.}{salt}
 Combine dry ingredients in a large mixing bowl.

 \ing[1]{cup}{butter}
 \ing[1\fr14]{cup}{white sugar}
 Melt the butter in a large mixing bowl. Add the sugar and mix until smooth.

 \ing[1]{}{egg}
 \ing[1]{tsp.}{vanilla}
 \ing[3]{tbsp.}{key lime juice}
 \ing[1]{tsp.}{lime extract}
 Mix the egg, vanilla, lime juice, and lime extract into the butter and sugar.

 \ing[6]{oz.}{white chocolate chips (\fr12 bag)}
 \ing[3]{}{graham crackers, crushed}
 Add the dry ingredients gradually, stirring to combine. Mix in the white chocolate chips and graham cracker chunks.

 \newstep Pour key lime bars into a greased baking dish. Bake 40 min.\ or until golden. Cool 45--60 min., slice, and serve.

\end{recipe}
