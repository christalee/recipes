\begin{recipe}{Peppermint Cookie Bars}{9\inch\X13\inch\ baking dish \hfill 350\0F \hfill 40 min.}{\textbf{Source:} Dustin, Brooklyn Community Chorus \hfill serves 24}
 \ing[2]{cups}{flour}
 \ing[1]{tsp.}{baking soda}
 \ing[\fr12]{tsp.}{salt}
 Combine flour, baking soda, and salt in a small bowl.

 \ing[1]{cup}{butter}
 \ing[\fr12]{cup}{white sugar}
 \ing[\fr12]{cup}{light brown sugar}
 Melt the butter in a large mixing bowl, then mix in the sugars until smooth.

 \ing[1]{}{egg}
 \ing[1]{tsp.}{vanilla}
 Mix the egg and vanilla into the butter and sugar until fluffy.

 \ing[12]{oz.}{Ghirardelli peppermint chunks}
 Gradually mix the dry ingredients into the wet ingredients. Stir in the peppermint chunks.

 \newstep Pour the peppermint bars into a greased baking dish. Bake for 40 min.\ or until golden. Let cool 45--60 min., slice and serve.
\end{recipe}
