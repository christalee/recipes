\begin{recipe}{Oreo Rice Krispie Treats}{9\inch\X13\inch baking dish \hfill Chill after mixing \hfill 20 min.}{\textbf{Source:}  Dustin, Brooklyn Community Choir \hfill serves 24}
 \ing[1]{cup}{butter}
 \ing[10]{oz.}{marshmallows (1 bag)}
 Melt the butter and marshmallows over low heat.

 \ing[3\fr12]{cups}{Rice Krispies}
 \ing[16]{}{Oreos}
 Add Rice Krispies and Oreos, and mix well.

 \newstep Spread mixture in a greased tray and let cool before slicing; refrigerate if needed.
\end{recipe}

\begin{recipe}{Mint Chocolate Chip Bars}{9\inch\X13\inch baking dish \hfill 350\0F \hfill 40 min.}{\textbf{Source:} Dustin, Brooklyn Community Choir \hfill serves 24}
 \ing[1]{cup}{butter}
 \ing[4]{oz.}{cream cheese}
 Over low heat, melt butter and cream cheese together.

 \ing[1]{box}{Pillsbury White Cake Mix (18 oz.)}
 \ing[1]{}{egg}
 \ing[\fr12]{tsp.}{peppermint extract}
 \ing[20--25]{drops}{(green food coloring)}
 In a medium bowl, add melted butter and cream cheese to the cake mix, egg, peppermint extract, and food coloring. Mix thoroughly.

 \ing[12]{oz.}{chocolate chips}
 Mix in chocolate chips. Pour mix into a greased baking dish.

 \newstep Bake until golden. Let cool 45--60 min.\ before slicing and serving.
\end{recipe}
